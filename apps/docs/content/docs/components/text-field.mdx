## Usage

```tsx
import { TextField } from '@aussieljk/frosted';

<TextField.Root>
  <TextField.Slot>
    <SearchIcon />
  </TextField.Slot>
  <TextField.Input placeholder="Search the docs…" />
</TextField.Root>;
```
